# Optional, platform-specific dependencies.
# Install with: pip install -r requirements-optional.txt
# Pip will only install lines whose environment markers match your system

# OQS (post-quantum) key encapsulation for utils/encryptor.py PersonalQuantumEncryptor.
# Install from GitHub; liboqs C library must be installed first (or is auto-installed at runtime).
# Full instructions: https://github.com/open-quantum-safe/liboqs-python/blob/main/README.md
git+https://github.com/open-quantum-safe/liboqs-python.git

# macOS (OS X): Foundation/Cocoa for optional keychain and native integration.
# Not required for basic keyring usage (utils/encryptor.py works without it).
pyobjc-framework-Cocoa; sys_platform == 'darwin'

# Linux: D-Bus for optional Secret Service keyring lock (utils/encryptor.py).
# Not required for basic keyring get/set.
dbus-python; sys_platform == 'linux'

# Cross-platform: Media key support for keyboard media control buttons (previous, play/pause, next).
# Required for the PySide6 version (app_qt.py) on macOS and Linux (Windows uses native Qt key detection).
# Used by lib/media_keys.py for global hotkey detection when the window doesn't have focus.
pynput

# ── TTS providers (install only the ones you intend to use) ──────────────────

# Kokoro: fast, high-quality named-voice TTS (Apache 2.0, ~82M params, ONNX).
# tts_provider = "kokoro" in config.json
kokoro-onnx

# F5-TTS: zero-shot voice cloning from a reference WAV (MIT).
# tts_provider = "f5tts" in config.json
f5-tts

# MaskGCT / Amphion: zero-shot voice cloning (non-commercial research license).
# tts_provider = "maskgct" in config.json
# Alternatively: git clone https://github.com/open-mmlab/Amphion && pip install -e Amphion/
amphion

# Piper: fast CPU-only TTS with 35+ language voices (Apache 2.0 / MIT).
# tts_provider = "piper" in config.json
# Voice .onnx files must be downloaded separately from:
#   https://huggingface.co/rhasspy/piper-voices
piper-tts

# Zonos: high-quality zero-shot voice cloning (Apache 2.0).
# tts_provider = "zonos" in config.json
# Requires espeak-ng on PATH; GPU recommended (6 GB+ VRAM).
# Also: pip install torch torchaudio  (if not already installed for Coqui)
git+https://github.com/Zyphra/Zonos.git
